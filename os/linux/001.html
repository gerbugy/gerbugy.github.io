--- layout: default title: 사용자, 그룹 관리 --- 1. 사용자
# 확인
$ cat /etc/passwd

# 추가
$ useradd mysql
$ useradd -g dba mysql # 그룹 지정

# 삭제
$ userdel mysql
$ userdel -r mysql # 사용자의 홈디렉토리도 삭제

# 수정
$ usermod mysql

# 비밀번호 설정
$ passwd mysql
2. 그룹
# 확인
$ groups
$ cat /etc/group

# 추가
$ groupadd dba
$ groupadd -g 567 dba # GID 지정

# 삭제
$ groupdel dba
